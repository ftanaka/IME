function [redc, u, ij] = custoDirecao(A, invB, c, n, m, I)
% Recebe:
% A: Matriz de restriÃ§Ãµes
% invB: Inversa da matriz B (colunas de A com indice bÃ¡sico)

% na qual o custo reduzido (custo na soluÃ§Ã£o encontrada a partir da direÃ§Ã£o) Ã© menor

function [I, invB] = atualizaBase(I, invB, u, imin, ij, m)
% Recebe:
% I: estrutura de indices bÃ¡sicos (I.b) e nÃ£o bÃ¡sicos (I.n)
% invB: Inversa da matriz B (colunas de A com indice bÃ¡sico)
% u: Componentes de indices bÃ¡sicos de uma direÃ§Ã£o viÃ¡vel
% imin: Indice de I.b
% ij: Indice de I.n
% m: nÃºmero de restriÃ§Ãµes
%
% Recalcula invB e I, de forma que x(I.b(imin)) saia da base e x(I.n(ij)) entre.
%
% Retorna:
% invB: B^-1 recalculada
% I: Ã­ndices recalculados
%
% Essa funÃ§Ã£o Ã© O(nm)
[I.b(imin), I.n(ij)] = deal(I.n(ij), I.b(imin));

end

function d = u2d(u, j, I)
